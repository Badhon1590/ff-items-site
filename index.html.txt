<html>
<head>
  <title>Free Fire Items</title>
  <style>
    body { font-family: Arial; background: #111; color: #fff; text-align: center; }
    .item { background: #222; margin: 10px; padding: 15px; border-radius: 8px; }
  </style>
</head>
<input type="text" id="search" placeholder="Search items by name or code" style="padding:8px; width: 300px; margin-bottom: 20px; border-radius: 5px; border:none;">

<body>
  <h1>Free Fire Items List</h1>
  <div id="items"></div>

  <script>
    fetch('items.json')
  .then(res => res.json())
  .then(data => {
    const itemsDiv = document.getElementById('items');
    const searchInput = document.getElementById('search');

    function displayItems(filteredItems) {
      let html = '';
      filteredItems.forEach(item => {
        html += `<div class="item">
                    <h2>${item.name}</h2>
                    <p>Code: ${item.code}</p>
                    <p>${item.description}</p>
                 </div>`;
      });
      itemsDiv.innerHTML = html;
    }

    displayItems(data);

    searchInput.addEventListener('input', () => {
      const query = searchInput.value.toLowerCase();
      const filtered = data.filter(item => 
        item.name.toLowerCase().includes(query) || 
        item.code.toLowerCase().includes(query)
      );
      displayItems(filtered);
    });
  });

                  </div>`;
        });
        document.getElementById('items').innerHTML = html;
      });
  </script>
</body>
</html>
