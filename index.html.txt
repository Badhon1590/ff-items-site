<html>
<head>
  <title>Free Fire Items</title>
  <style>
    body { font-family: Arial; background: #111; color: #fff; text-align: center; }
    input#search {
      padding: 8px; width: 300px; margin: 20px auto; border-radius: 5px; border: none; display: block;
      font-size: 16px;
    }
    .item {
      background: #222; margin: 10px auto; padding: 15px; border-radius: 8px; max-width: 400px; text-align: left;
    }
  </style>
</head>
<body>
  <h1>Free Fire Items List</h1>
  <input type="text" id="search" placeholder="Search items by name or code">

  <div id="items"></div>

  <script>
    fetch('items.json')
      .then(res => res.json())
      .then(data => {
        const itemsDiv = document.getElementById('items');
        const searchInput = document.getElementById('search');

        function displayItems(items) {
          if(items.length === 0) {
            itemsDiv.innerHTML = '<p>No items found</p>';
            return;
          }
          let html = '';
          items.forEach(item => {
            html += `<div class="item">
                      <h2>${item.name}</h2>
                      <p><strong>Code:</strong> ${item.code}</p>
                      <p>${item.description}</p>
                    </div>`;
          });
          itemsDiv.innerHTML = html;
        }

        displayItems(data);

        searchInput.addEventListener('input', () => {
          const query = searchInput.value.toLowerCase();
          const filtered = data.filter(item => 
            item.name.toLowerCase().includes(query) || 
            item.code.toLowerCase().includes(query)
          );
          displayItems(filtered);
        });
      })
      .catch(err => {
        document.getElementById('items').innerHTML = '<p>Error loading items.</p>';
        console.error(err);
      });
  </script>
</body>
</html>
